type Coin = HEADS | TAILS
let getTT (p:float*float) = 
    let flips = [HEADS;TAILS]
    let getP coin = if coin = HEADS then fst(p) else snd(p)
    [
        for i in flips do
            for j in flips do
                for k in flips ->
                    (i,j,k,(getP(i)*getP(j)*getP(k)))
    ] 

let tt = getTT (0.6, 0.4)
