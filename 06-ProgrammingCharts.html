let getDist r l =     
    let counts = Array.init (r|>Seq.length) (fun _ -> 0)

    let rec loop r l =   
        match r with
        | [] -> ()
        | (mn,mx)::tr -> 
            let i = (r|>Seq.length) - 1
            match l with
            | [] -> ()
            | h::t ->
                if h >= mn && h < mx then
                    counts.[i] <- counts.[i] + 1
                    printfn "%f %f %f %A %A" h mn mx counts r
                    loop r t
                else                     
                    counts.[i-1] <- counts.[i-1] + 1
                    printfn "%f %f %f %A %A" h mn mx counts r
                    loop tr t

    loop r (l |> List.sort)
    (r 
        |> Seq.map (fun v -> 
            if snd(v) = System.Double.MaxValue then sprintf ">%.0f" (fst(v)) else sprintf "%.0f-%.0f" (fst(v)) (snd(v)))
            , (counts |> Array.rev)) 
        ||> Seq.zip

let getGroups n l =
    let ng = (l |> Seq.length)/n
    let arr = l |> Seq.sort |> Array.ofSeq

    [
        for i in 0..(n-1) ->
            [ for j in 0..(ng-1) -> arr.[(i*ng) + j]]            
    ]
